클라우드 자동 백업 설정
마감할 때마다 JSON 파일이 본인 구글드라이브에 자동 업로드됩니다. 한 번만 세팅하면 이후는 완전 자동이에요.
연결됨 · 마지막 업로드 18시간 전
웹앱 URL
아래 "세팅 가이드 보기" 를 따라 URL 하나를 만드신 후 여기에 붙여넣으세요.
세팅 가이드 보기 (처음 1번만, 약 5분)
  1. script.google.com/home/projects/create 접속
  2. 기본 코드(function myFunction() {})를 모두 지우고 아래 눌러 복사한 뒤 붙여넣기
  3. 상단 💾 저장 아이콘 클릭 → 프로젝트 이름 "정산백업" 입력
  4. 우측 상단 파란색 "배포" 버튼 → "새 배포"
  5. 좌측 ⚙️ 톱니바퀴 아이콘 → "웹 앱" 선택
  6. 설정:
      • 설명: 아무거나
      • 실행: "나"
      • 액세스 권한: "모든 사용자" ← 중요!
  7. "배포" 클릭 → "액세스 승인" → 본인 구글계정 선택
  8. ⚠️ "Google에서 확인하지 않은 앱" 경고 화면: "고급" → "정산백업(안전하지 않음)(으)로 이동" → "허용"
  9. 생성된 웹 앱 URL 복사 (https://script.google.com/macros/s/.../exec)
  10. 위 URL 입력란에 붙여넣고 "연결 테스트"
⚠️ 액세스 권한이 반드시 "모든 사용자"여야 합니다.
"나만"으로 설정하면 브라우저가 GAS에 접근할 수 없어 업로드 실패합니다.
URL이 노출돼도 본인이 작성한 코드(JSON 파일 저장만)만 실행되니 안전해요.
이미 배포하신 경우:
아래 코드가 이전 버전에서 업데이트되었어요. 기존 프로젝트의 코드를 모두 지우고 새 코드를 다시 붙여넣은 뒤, "배포 → 배포 관리 → 연필 ✏️ → 새 버전 → 배포"로 재배포해주세요. URL은 그대로 유지됩니다.
붙여넣을 코드:
/**
 * 일일정산 자동 백업 - Google Apps Script
 * 쓰기(doPost) + 읽기(doGet) 모두 지원
 */

const FOLDER_NAME = '정산_백업';
const SYNC_FILENAME = '전체백업_최신.json';

function doPost(e) {
  try {
    let data;
    if (e.parameter && e.parameter.payload) {
      data = JSON.parse(e.parameter.payload);
    } else if (e.postData && e.postData.contents) {
      data = JSON.parse(e.postData.contents);
    } else {
      throw new Error('No payload received');
    }

    const folder = getOrCreateFolder(FOLDER_NAME);

    // Replace existing file with same name
    const existing = folder.getFilesByName(data.filename);
    while (existing.hasNext()) {
      folder.removeFile(existing.next());
    }

    const file = folder.createFile(
      data.filename,
      JSON.stringify(data.content, null, 2),
      'application/json'
    );

    return HtmlService.createHtmlOutput(
      '<html><body>OK: ' + file.getName() + '</body></html>'
    );
  } catch (err) {
    return HtmlService.createHtmlOutput(
      '<html><body>ERROR: ' + err.toString() + '</body></html>'
    );
  }
}

function doGet(e) {
  try {
    // ?action=sync to fetch latest backup
    if (e && e.parameter && e.parameter.action === 'sync') {
      const folder = getOrCreateFolder(FOLDER_NAME);
      const files = folder.getFilesByName(SYNC_FILENAME);

      if (!files.hasNext()) {
        return ContentService.createTextOutput(JSON.stringify({
          ok: true,
          empty: true,
          message: '동기화 파일 없음 - 아직 백업된 적 없음'
        })).setMimeType(ContentService.MimeType.JSON);
      }

      const file = files.next();
      const content = file.getBlob().getDataAsString('UTF-8');

      return ContentService.createTextOutput(JSON.stringify({
        ok: true,
        lastModified: file.getLastUpdated().toISOString(),
        content: JSON.parse(content)
      })).setMimeType(ContentService.MimeType.JSON);
    }

    // Default: service info
    return ContentService.createTextOutput(JSON.stringify({
      ok: true,
      service: '일일정산 백업',
      version: 'v2',
      time: new Date().toISOString()
    })).setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({
      ok: false,
      error: err.toString()
    })).setMimeType(ContentService.MimeType.JSON);
  }
}

function getOrCreateFolder(name) {
  const folders = DriveApp.getFoldersByName(name);
  if (folders.hasNext()) return folders.next();
  return DriveApp.createFolder(name);
}
